f_pair_from_ticker() =>
    // 1) 先嘗試用 basecurrency / currency（比較穩）
    string b = str.upper(syminfo.basecurrency)
    string q = str.upper(syminfo.currency)

    string pair6 = ""
    bool bq_ok = (str.length(b) == 3 and str.length(q) == 3)

    if bq_ok
        pair6 := b + q
    else
        // 2) 退回用 ticker 最後 6 碼
        string tk = str.upper(syminfo.ticker)
        if str.length(tk) >= 6
            pair6 := str.substring(tk, str.length(tk) - 6, str.length(tk))
            b := str.substring(pair6, 0, 3)
            q := str.substring(pair6, 3, 6)
        else
            pair6 := ""

    // 3) 28 組白名單（只允許這些 pair）
    bool ok =
         pair6 == "EURUSD" or pair6 == "GBPUSD" or pair6 == "AUDUSD" or pair6 == "NZDUSD" or pair6 == "USDJPY" or pair6 == "USDCHF" or pair6 == "USDCAD" or
         pair6 == "EURJPY" or pair6 == "GBPJPY" or pair6 == "AUDJPY" or pair6 == "NZDJPY" or pair6 == "CADJPY" or pair6 == "CHFJPY" or
         pair6 == "EURGBP" or pair6 == "EURAUD" or pair6 == "EURNZD" or pair6 == "EURCAD" or pair6 == "EURCHF" or
         pair6 == "GBPAUD" or pair6 == "GBPNZD" or pair6 == "GBPCAD" or pair6 == "GBPCHF" or
         pair6 == "AUDNZD" or pair6 == "AUDCAD" or pair6 == "AUDCHF" or
         pair6 == "NZDCAD" or pair6 == "NZDCHF" or
         pair6 == "CADCHF"

    // 4) 不在白名單就回傳空字串 -> pair_supported 會變 false
    if not ok
        b := ""
        q := ""

    [b, q]

var box[] rw_sl_boxes = array.new_box()
var box[] rw_tp_boxes = array.new_box()

f_rw_gc(_maxPairs) =>
    while array.size(rw_sl_boxes) > _maxPairs
        box b1 = array.shift(rw_sl_boxes)
        if not na(b1)
            box.delete(b1)
    while array.size(rw_tp_boxes) > _maxPairs
        box b2 = array.shift(rw_tp_boxes)
        if not na(b2)
            box.delete(b2)

bool rw_ready = rw_show_boxes and trade_fired and (rw_hN >= 1)

var int last_box_bar = na
bool can_draw_this_bar = na(last_box_bar) or (bar_index != last_box_bar)

if rw_ready and can_draw_this_bar
    float entry = entry_ref
    int left_idx = bar_index
    int right_idx = bar_index + (rw_hN - 1)

    if rw_dir == -1
        float sl = entry + slPts_exec
        float tp = entry - tpPts_exec
        box b_sl = box.new(left=left_idx, top=math.max(entry, sl), right=right_idx, bottom=math.min(entry, sl), bgcolor=rw_sl_col, border_color=color.new(color.black, 100), force_overlay=true)
        box b_tp = box.new(left=left_idx, top=math.max(entry, tp), right=right_idx, bottom=math.min(entry, tp), bgcolor=rw_tp_col, border_color=color.new(color.black, 100), force_overlay=true)
        array.push(rw_sl_boxes, b_sl), array.push(rw_tp_boxes, b_tp)
        f_rw_gc(rw_max_pairs), last_box_bar := bar_index
    else if rw_dir == 1
        float sl = entry - slPts_exec
        float tp = entry + tpPts_exec
        box b_sl = box.new(left=left_idx, top=math.max(entry, sl), right=right_idx, bottom=math.min(entry, sl), bgcolor=rw_sl_col, border_color=color.new(color.black, 100), force_overlay=true)
        box b_tp = box.new(left=left_idx, top=math.max(entry, tp), right=right_idx, bottom=math.min(entry, tp), bgcolor=rw_tp_col, border_color=color.new(color.black, 100), force_overlay=true)
        array.push(rw_sl_boxes, b_sl), array.push(rw_tp_boxes, b_tp)
        f_rw_gc(rw_max_pairs), last_box_bar := bar_index
